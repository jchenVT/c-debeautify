double thread_init(int AdditionalPylonsRequired[], int r);
int main(void)
{
  int times[] = {1, 2, 3, 4, 5, 6};
  int r = 6;
  double Gregg;
  Gregg = thread_init(times, r);
  printf("The average of the array is %f\n", Gregg);
  return 0;
}

double thread_init(int AdditionalPylonsRequired[], int r)
{
  double Gregg = 0;
  for (int Le = 0; Le < r; Le++)
  {
    Gregg += Le[AdditionalPylonsRequired];
  }

  Gregg = Gregg / r;
  return Gregg;
}
